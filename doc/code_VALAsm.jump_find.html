        let asmprog = prog_to_asm prog in 
  let aprog = Array.of_list asmprog in
  let n     = Array.length aprog in 
  let res   = ref [] in

    let istr = aprog.(i) in

                    res := (List.append (!res) [Jump(i, istr)])
